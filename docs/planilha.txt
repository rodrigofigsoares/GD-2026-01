import pandas as pd

arquivo = "gemeo_digital_painel_solar_horarios_2.xlsx"

df = pd.read_excel(
    arquivo,
    sheet_name="Dados_Horarios"
)

localizacao = Localizacao(
    df.loc[0, "ID_LOCALIZACAO"],
    df.loc[0, "LAT"],
    df.loc[0, "LON"]
)

sistema = SistemaSolar("Painel 60 Células")

gemeo = GemeoDigital(
    localizacao,
    sistema
)

for _, linha in df.iterrows():

    data_hora = datetime(
        int(linha["Ano"]),
        int(linha["Mes"]),
        int(linha["Dia"]),
        int(linha["Hora"])
    )

    medicao = MedicaoHoraria(
        data_hora=data_hora,
        temperatura=linha["T2M_C"],
        umidade=linha["RH2M_%"],
        vento=linha["WS10M_ms"],
        irradiancia=linha["GHI_Wh_m2"]
    )

    estado = EstadoPainel(
        temperatura_celula=linha["60CEL_Tcel_C"],
        potencia_dc=linha["60CEL_Pdc_W"],
        potencia_real=linha["60CEL_Preal_W"],
        fator_temperatura=linha["60CEL_FatorTemp"]
    )

    gemeo.atualizar(
        medicao,
        estado
    )

gemeo.exibir_status()
