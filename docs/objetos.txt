class Localizacao:
    def __init__(self, id_localizacao, lat, lon):
        self.id = id_localizacao
        self.latitude = lat
        self.longitude = lon

class PainelSolar:
    def __init__(self, tipo):
        self.tipo = tipo
        self.potencia_real = 0
        self.temperatura_celula = 0

class Medicao:
    def __init__(self, data_hora, temperatura, umidade,
                 vento, irradiancia):
        self.data_hora = data_hora
        self.temperatura = temperatura
        self.umidade = umidade
        self.vento = vento
        self.irradiancia = irradiancia

class GemeoDigital:
    def __init__(self, localizacao, painel):
        self.localizacao = localizacao
        self.painel = painel
        self.historico = []

    def atualizar_estado(self, medicao,
                         temp_celula,
                         potencia_real):
        self.painel.temperatura_celula = temp_celula
        self.painel.potencia_real = potencia_real
        self.historico.append(medicao)
